cut <- 100
nino34.grid <-
  with(nino34,
     outer(values, seq(min(values), max(values), length=cut), function(X,Y) {
         (X >= 0 & X > Y & Y > 0) | (X <= 0 & X < Y & Y < 0)
       }))

nino34.grid <- t(apply(nino34.grid, 1, function(x) {
  x[x == F] <- NA;
  x*seq(min(nino34$values), max(nino34$values), length=cut)
}))
